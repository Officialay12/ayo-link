API Example 

const shortenUrl = async () => {
    const response = await fetch('https://api.ayo.link/shorten', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            url: 'https://example.com/long-url',
            customSlug: 'my-custom-link',
            apiKey: 'ayo_your_api_key_here'
        })
    });

    const data = await response.json();
    console.log(data.shortUrl); // https://ayo.link/abc123
};
